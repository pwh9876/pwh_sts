
CREATE TABLE TS_TBL_TRD (
    TRD_DT CHAR(20) NOT NULL,
    ACNT_NO CHAR(9) NOT NULL,
    TRD_NO INT NOT NULL,
    AMT INT ,
    CMSN INT,
    CANC_YN CHAR(1),
    PRIMARY KEY(TRD_DT, ACNT_NO, TRD_NO)
);


CREATE TABLE TS_TBL_MGMT_BRN(
    MGMT_BRN_CODE CHAR(1) PRIMARY KEY ,
    MGMT_BRN_NM VARCHAR(20) 
);


CREATE TABLE TS_TBL_ACNT(
    ACNT_NO CHAR(20) PRIMARY KEY,
    ACNT_NM VARCHAR(20) ,
    MGMT_BRN_CODE CHAR(1)
);
